create database spareservation;

//CUSTOMER
CREATE TABLE CUSTOMER(ID INT NOT NULL PRIMARY KEY AUTO_INCREMENT, Last_Name VARCHAR(45) NOT NULL, First_Name VARCHAR(45) NOT NULL, Contact_Number VARCHAR(15) NOT NULL);

//ROOMS
CREATE TABLE ROOMS(ID INT NOT NULL PRIMARY KEY AUTO_INCREMENT, Room_Number VARCHAR(2) NOT NULL);

//EMPLOYEE
CREATE TABLE EMPLOYEE(ID INT NOT NULL PRIMARY KEY AUTO_INCREMENT, Last_Name VARCHAR(45) NOT NULL, First_Name VARCHAR(45) NOT NULL, Position VARCHAR(45) NOT NULL);

//SERVICES
CREATE TABLE SERVICES(ID INT NOT NULL PRIMARY KEY AUTO_INCREMENT, Service_Name VARCHAR(45) NOT NULL, Price DECIMAL NOT NULL);

//BOOKING
CREATE TABLE BOOKING(ID INT NOT NULL PRIMARY KEY AUTO_INCREMENT, Booking_Type VARCHAR(15) NOT NULL, time_start TIME NOT NULL, time_end TIME NOT NULL, date_received DATE NOT NULL, duration TIME NOT NULL, Rooms_ID INT,Customer_ID INT, Employee_ID INT, FOREIGN KEY (Rooms_ID) REFERENCES Rooms(ID), FOREIGN KEY (Customer_ID) REFERENCES customer(ID), FOREIGN KEY (Employee_ID) REFERENCES employee(ID));

//BOOKING_DETAILS
CREATE TABLE booking_details(ID INT NOT NULL PRIMARY KEY AUTO_INCREMENT, Booking_ID INT, Employee_ID INT, Services_ID INT, FOREIGN KEY (Booking_ID) REFERENCES booking(ID), FOREIGN KEY (Employee_ID) REFERENCES employee(ID), FOREIGN KEY (Services_ID) REFERENCES services(ID));
